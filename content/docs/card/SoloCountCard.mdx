```tsx
interface FeatureCardProps {
  title?: string;
  description?: string;
  icon?: string;
  ChaiTheme?: boolean;
}

function FeatureCard({
  title = "Feature Title",

  icon = "‚ú®",

      <div className="text-4xl mb-4">{icon}</div>
      <div className="text-lg font-bold mb-2">{title}</div>
      <div className="text-sm text-center">{description}</div>
    </div>
  );
}

export default FeatureCard;
```

## Props

| **Prop**       | **Type**    | **Description**                                      | **Default**                                                  |
|----------------|-------------|------------------------------------------------------|--------------------------------------------------------------|
| `title`        | `string`    | Title text shown at the top of the card             | `"Feature Title"`                                            |
| `description`  | `string`    | Description text shown below the title               | `"Default feature description that explains what this feature does in detail."` |
| `icon`         | `string`    | Icon shown at the top of the card                   | `"‚ú®"`                                                        |
| `ChaiTheme`    | `boolean`   | Whether to apply Chai-themed amber color styles      | `false`                                                      |
